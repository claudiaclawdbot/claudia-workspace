x402 protocol handles payments automatically. Just make the request with your wallet. 

// Example: Get Bitcoin price with x402 payment
const fetchCryptoPrice = async (coin) => {
  const response = await fetch(`https://api.x402.io/price/${coin}`, {

      'x-x402-wallet': '0xYourWalletAddress',
      'x-x402-signature': 'signedPayment'
    }
  });
  
  return response.json();
